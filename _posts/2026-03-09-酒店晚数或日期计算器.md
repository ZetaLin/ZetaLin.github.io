<script>
    // 使用自执行函数避免全局污染
    (function() {
        const inInput = document.getElementById('inDate');
        const outInput = document.getElementById('outDate');
        const res = document.getElementById('nightRes');

        // 初始化日期为今天
        const today = new Date().toISOString().split('T')[0];
        inInput.value = today;

        function calc() {
            const d1 = new Date(inInput.value);
            const d2 = new Date(outInput.value);
            if (!isNaN(d1) && !isNaN(d2)) {
                const diff = Math.ceil((d2 - d1) / (1000 * 60 * 60 * 24));
                res.innerText = diff > 0 ? diff : 0;
            }
        }

        window.setDays = function(days) {
            const start = new Date(inInput.value);
            start.setDate(start.getDate() + days);
            outDate.value = start.toISOString().split('T')[0];
            calc();
        };

        inInput.addEventListener('change', calc);
        outInput.addEventListener('change', calc);
    })();
</script>
