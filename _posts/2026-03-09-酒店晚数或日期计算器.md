<script>
(function() {
    // 功能一逻辑
    const s1 = document.getElementById('start_1');
    const e1 = document.getElementById('end_1');
    const r1 = document.getElementById('res_1');

    function calcInterval() {
        const d1 = new Date(s1.value);
        const d2 = new Date(e1.value);
        if(!isNaN(d1) && !isNaN(d2)) {
            const diff = Math.ceil((d2 - d1) / (86400000));
            r1.innerText = diff >= 0 ? diff : 0;
        }
    }

    // 功能二逻辑
    const s2 = document.getElementById('start_2');
    const d2 = document.getElementById('days_2');
    const resDate = document.getElementById('res_date_2');
    const resWeek = document.getElementById('res_week_2');

    // 初始化功能二日期为今天
    const today = new Date().toISOString().split('T')[0];
    s2.value = today;

    function calcNewDate() {
        const baseDate = new Date(s2.value);
        const addDays = parseInt(d2.value) || 0;
        if(!isNaN(baseDate)) {
            baseDate.setDate(baseDate.getDate() + addDays);
            const y = baseDate.getFullYear();
            const m = baseDate.getMonth() + 1;
            const d = baseDate.getDate();
            const week = ['星期日','星期一','星期二','星期三','星期四','星期五','星期六'][baseDate.getDay()];
            
            resDate.innerText = `${y}年${m}月${d}日`;
            resWeek.innerText = week;
        }
    }

    window.quickAdd = function(n) {
        d2.value = n;
        calcNewDate();
    }

    // 绑定事件
    [s1, e1].forEach(el => el.addEventListener('change', calcInterval));
    [s2, d2].forEach(el => el.addEventListener('input', calcNewDate));

    // 初始执行一次
    calcInterval();
    calcNewDate();
})();
</script>
