<script>
(function() {
    const today = new Date().toISOString().split('T')[0];

    // --- 逻辑一：算晚数 ---
    const inInp = document.getElementById('inDate');
    const outInp = document.getElementById('outDate');
    const nightRes = document.getElementById('nightRes');

    inInp.value = today;
    
    function calcNights() {
        const d1 = new Date(inInp.value);
        const d2 = new Date(outInp.value);
        if (!isNaN(d1) && !isNaN(d2)) {
            const diff = Math.ceil((d2 - d1) / 86400000);
            nightRes.innerText = diff > 0 ? diff : 0;
        }
    }

    // --- 逻辑二：算日期 ---
    const baseInp = document.getElementById('baseDate');
    const offsetInp = document.getElementById('offsetDays');
    const dateRes = document.getElementById('newDateRes');
    const weekRes = document.getElementById('newWeekRes');

    baseInp.value = today;

    function calcNewDate() {
        const base = new Date(baseInp.value);
        const days = parseInt(offsetInp.value) || 0;
        if (!isNaN(base)) {
            base.setDate(base.getDate() + days);
            const y = base.getFullYear();
            const m = base.getMonth() + 1;
            const d = base.getDate();
            const weeks = ['星期日','星期1','星期二','星期三','星期四','星期五','星期六'];
            
            dateRes.innerText = `${y}年${m}月${d}日`;
            weekRes.innerText = weeks[base.getDay()];
        }
    }

    window.quickOffset = function(n) {
        offsetInp.value = n;
        calcNewDate();
    };

    // 绑定事件
    inInp.onchange = calcNights;
    outInp.onchange = calcNights;
    baseInp.onchange = calcNewDate;
    offsetInp.oninput = calcNewDate;

    // 初始计算一次
    calcNewDate();
})();
</script>>
